use Canada;

CREATE TABLE state_alpha_code (
  alpha_code char(2) NOT NULL,
  description NVARCHAR(50) NOT NULL,
  country NVARCHAR(50) NOT NULL ,
  PRIMARY KEY  (alpha_code)
);

CREATE TABLE state_region (
  alpha_code char(2) NOT NULL,
  region NVARCHAR(50) NOT NULL,
  PRIMARY KEY  (alpha_code,region),
  FOREIGN KEY (alpha_code) REFERENCES state_alpha_code(alpha_code)
);

CREATE TABLE state_city (
  alpha_code char(2) NOT NULL,
  city NVARCHAR(50) NOT NULL,
  PRIMARY KEY  (alpha_code,city),
  FOREIGN KEY (alpha_code) REFERENCES state_alpha_code(alpha_code)
);
